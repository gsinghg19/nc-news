import { useState } from "react";

const Counter = () => {
  const [vote, setVote] = useState(0);

  const increaseVote = () => {
    setVote(function (currentVote) {
      return currentVote + 1;
    });
  };

  const resetVote = () => {
    setVote(0);
  };

  return (
    <div>
      <p>Number of Votes{vote}</p>
      <button onClick={increaseVote}>increase</button>
      <button onClick={resetVote}>reset</button>
    </div>
  );
};

export default Counter;
