import { useState } from "react";

const UseVote = () => {
  const [vote, setVote] = useState(0);

  const increaseVote = () => {
    setVote(function (currentVote) {
      return currentVote + 1;
    });
  };

  const resetVote = () => {
    setVote(0);
  };

  return { vote, increaseVote, resetVote };
};

export default UseVote;
